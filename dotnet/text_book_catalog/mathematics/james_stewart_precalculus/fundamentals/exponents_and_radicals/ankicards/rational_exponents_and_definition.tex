        \makecell{
            $ a^{\frac{1}{n}} = \sqrt[n]{a} $\\
            \\
            For any rational exponent $m/n$ in lowest terms,\\ 
            where $m$ and $n$ are integers and $n > 0$, we define: \\
            $ a^{m/n} = \left(\sqrt[n]a\right)^m \text{ or equivalently } a^{m/n} = \sqrt[n]a^m $\\
            If $n$ is even, then we require that $a \geq 0$.
        }
        \\
        \bottomrule
    \end{tabularx}
\end{small}
